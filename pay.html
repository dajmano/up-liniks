<head>

  <meta charset="UTF-8">

  <title>Redirecting to UPI</title>

  <meta name="viewport" content="width=device-width, initial-scale=1">

  <script>

    function getParam(name) {

      const url = new URL(window.location.href);

      return encodeURIComponent(url.searchParams.get(name) || '');

    }


    const pa = getParam("pa");

    const pn = getParam("pn");

    const am = getParam("am");

    const tn = getParam("tn");


    const upiUrl = `upi://pay?pa=${pa}&pn=${pn}&am=${am}&cu=INR&tn=${tn}`;

    window.location.href = upiUrl;

  </script>

</head>

  <p>Redirecting to UPI App... <br>

     If not redirected, <a id="link" href="#">Click here</a></p>
